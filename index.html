1. Reading and Writing Characters

Functions

  • getchar() → read one character
  • putchar() → print one character

Example

#include <stdio.h>

int main() {
    char ch;

    printf("Enter a character: ");
    ch = getchar();

    printf("You entered: ");
    putchar(ch);

    return 0;
}

⚠️ Case: Handling newline issue

char ch1 = getchar();
char ch2 = getchar();  // may read '\n'

Fix:

getchar(); // consume newline

2. Reading and Writing Strings

Functions

  • scanf("%s", str) → stops at space
  • fgets() → reads full line

Case 1: Using scanf

#include <stdio.h>

int main() {
    char name[20];

    printf("Enter name: ");
    scanf("%s", name);

    printf("Name: %s\n", name);
}

Input: Sam Prasanth

Output: Sam (stops at space)

Case 2: Using fgets (recommended)

#include <stdio.h>

int main() {
    char name[50];

    printf("Enter name: ");
    fgets(name, sizeof(name), stdin);

    printf("Name: %s", name);
}

3. Formatted Console I/O

printf() → output formatting

scanf() → input formatting

printf example

#include <stdio.h>

int main() {
    int a = 10;
    float b = 3.14159;

    printf("Integer: %d\n", a);
    printf("Float: %.2f\n", b);

    return 0;
}

scanf example

#include <stdio.h>

int main() {
    int age;
    float salary;

    printf("Enter age and salary: ");
    scanf("%d %f", &age, &salary);

    printf("Age: %d, Salary: %.2f\n", age, salary);

    return 0;
}

⚠️ Common mistake

scanf("%d", age);   //  wrong
scanf("%d", &age);  //  correct

4. Statements in C

🔸 4.1 True and False in C

  • 0 → FALSE
  • non-zero → TRUE
#include <stdio.h>

int main() {
    if (5) {
        printf("True\n");
    }

    if (0) {
        printf("False\n");
    }
}

5. Selection Statements (Decision Making)

if

if (condition) {
    // code
}

if-else

if (a > b) {
    printf("A is bigger");
} else {
    printf("B is bigger");
}

else-if ladder

if (marks >= 90)
    printf("A");
else if (marks >= 75)
    printf("B");
else
    printf("C");

switch case

#include <stdio.h>

int main() {
    int day = 2;

    switch(day) {
        case 1: printf("Mon"); break;
        case 2: printf("Tue"); break;
        default: printf("Invalid");
    }
}

6. Iteration Statements (Loops)

for loop

for (int i = 1; i <= 5; i++) {
    printf("%d ", i);
}

while loop

int i = 1;
while (i <= 5) {
    printf("%d ", i);
    i++;
}

do-while loop

int i = 1;
do {
    printf("%d ", i);
    i++;
} while (i <= 5);

Runs at least once

7. Jump Statements

break

for (int i = 1; i <= 5; i++) {
    if (i == 3)
        break;
    printf("%d ", i);
}

continue

for (int i = 1; i <= 5; i++) {
    if (i == 3)
        continue;
    printf("%d ", i);
}

goto (rare, avoid)

#include <stdio.h>

int main() {
    goto label;

    printf("Skipped\n");

label:
    printf("Jumped here\n");
}

8. Expression Statements

Any expression ending with ;

int a = 5;
a = a + 2;
printf("%d", a);

9. Block Statements

Group of statements inside {}

if (1) {
    int x = 10;
    printf("%d", x);
}

Final Summary (Important for Interview)

  • getchar/putchar → single character
  • scanf vs fgets → word vs full line
  • printf → formatted output
  • 0 = false, non-zero = true
  • Selection → if, switch
  • Iteration → for, while, do-while
  • Jump → break, continue, goto
  • Expression → statements with ;
  • Block → { }
